8. Palindrome Check 

      
var Palin_Check = (str) =>
{
	let i=0
         let j= str.length-1
         let count=0
         
         while(i< j){
           if(str[i] !== str[j]){
             count = 1;
             break;
           }
           i++
           j--
         }
         
         if(count ===1){
           return "False"
         }else{
           return "True"
         }
}

var Reverse_String = (str) => 
{
    const tempArr = []
        let reverseStr = ''
        for(let i= str.length-1; i>=0; i--){
          tempArr.push(str.charAt(i))
        }
        
        for(let i=0; i< tempArr.length; i++){
          reverseStr += tempArr[i]
        }
        return reverseStr 
};
